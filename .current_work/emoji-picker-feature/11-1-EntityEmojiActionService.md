## Task 1 — Create emoji action repository

Low-level CRUD operations on the `entity_emoji_actions` table. Follows the same pattern as `emotion_state.ts`:

import { getDatabase } from '../connection';
import { withTransaction } from '../transaction';

function mapRowToAction(row: any): EmojiAction {
  return {
    id: row.id,
    entityId: row.entity_id,
    emojiNative: row.emoji_native,
    emotionEffect: row.emotion_effect ? JSON.parse(row.emotion_effect) : null,
    metabolismVector: row.metabolism_vector ? JSON.parse(row.metabolism_vector) : null,
    substitutionText: row.substitution_text,
    autoGenerated: row.auto_generated === 1,
    isDefault: row.is_default === 1,
    createdAt: new Date(row.created_at),
    updatedAt: new Date(row.updated_at),
  };
}

/**
 * Get all emoji actions for a specific entity.
 * Returns a Map<emojiNative, EmojiAction> for O(1) lookup during typing.
 */
export async function getEmojiActionsForEntity(
  entityId: string,
): Promise<Map<string, EmojiAction>> {
  const db = getDatabase();
  const [result] = await db.executeSql(
    'SELECT * FROM entity_emoji_actions WHERE entity_id = ? ORDER BY emoji_native',
    [entityId],
  );

  const map = new Map<string, EmojiAction>();
  for (let i = 0; i < result.rows.length; i++) {
    const action = mapRowToAction(result.rows.item(i));
    map.set(action.emojiNative, action);
  }
  return map;
}

/**
 * Get a single emoji action by entity + emoji.
 * Returns null if no action is defined for this emoji.
 */
export async function getEmojiAction(
  entityId: string,
  emojiNative: string,
): Promise<EmojiAction | null> {
  const db = getDatabase();
  const [result] = await db.executeSql(
    'SELECT * FROM entity_emoji_actions WHERE entity_id = ? AND emoji_native = ?',
    [entityId, emojiNative],
  );

  if (result.rows.length === 0) return null;
  return mapRowToAction(result.rows.item(0));
}

/**
 * Get all emoji actions for a specific entity as an array.
 * Used by the action editor screen to display the full list.
 */
export async function getAllEmojiActionsArray(
  entityId: string,
): Promise<EmojiAction[]> {
  const db = getDatabase();
  const [result] = await db.executeSql(
    'SELECT * FROM entity_emoji_actions WHERE entity_id = ? ORDER BY is_default DESC, emoji_native',
    [entityId],
  );

  const actions: EmojiAction[] = [];
  for (let i = 0; i < result.rows.length; i++) {
    actions.push(mapRowToAction(result.rows.item(i)));
  }
  return actions;
}

/**
 * Create a new emoji action. Fails if one already exists for this entity+emoji
 * (use upsertEmojiAction for insert-or-replace semantics).
 */
export async function createEmojiAction(
  action: Omit<EmojiAction, 'createdAt' | 'updatedAt'>,
): Promise<void> {
  const db = getDatabase();
  const now = new Date().toISOString();

  await db.executeSql(
    `INSERT INTO entity_emoji_actions (
      id, entity_id, emoji_native, emotion_effect, metabolism_vector,
      substitution_text, auto_generated, is_default, created_at, updated_at
    ) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?)`,
    [
      action.id,
      action.entityId,
      action.emojiNative,
      action.emotionEffect ? JSON.stringify(action.emotionEffect) : null,
      action.metabolismVector ? JSON.stringify(action.metabolismVector) : null,
      action.substitutionText,
      action.autoGenerated ? 1 : 0,
      action.isDefault ? 1 : 0,
      now,
      now,
    ],
  );
}

/**
 * Upsert an emoji action. Inserts if not exists, replaces if it does.
 * Used for both creating new actions and updating existing ones.
 */
export async function upsertEmojiAction(
  action: Omit<EmojiAction, 'createdAt' | 'updatedAt'>,
): Promise<void> {
  const db = getDatabase();
  const now = new Date().toISOString();

  await db.executeSql(
    `INSERT OR REPLACE INTO entity_emoji_actions (
      id, entity_id, emoji_native, emotion_effect, metabolism_vector,
      substitution_text, auto_generated, is_default, created_at, updated_at
    ) VALUES (?, ?, ?, ?, ?, ?, ?, ?, ?, ?)`,
    [
      action.id,
      action.entityId,
      action.emojiNative,
      action.emotionEffect ? JSON.stringify(action.emotionEffect) : null,
      action.metabolismVector ? JSON.stringify(action.metabolismVector) : null,
      action.substitutionText,
      action.autoGenerated ? 1 : 0,
      action.isDefault ? 1 : 0,
      now,
      now,
    ],
  );
}

/**
 * Delete a specific emoji action by ID.
 */
export async function deleteEmojiAction(actionId: string): Promise<void> {
  const db = getDatabase();
  await db.executeSql(
    'DELETE FROM entity_emoji_actions WHERE id = ?',

/**
 * Delete all emoji actions for a specific entity.

 * Count emoji actions for a specific entity.

  const db = getDatabase();
  const [result] = await db.executeSql(
    'SELECT COUNT(*) as count FROM entity_emoji_actions WHERE entity_id = ?',

// ============================================================================
// Effect Aggregation
// ============================================================================

/**
 * Aggregate multiple EmotionEffects for the same emotion.
 *
 * Aggregation rule: For each emotion, the result is clamped between the
 * most-negative individual effect and the most-positive individual effect.
 *
 *   pos_ceiling = max of all positive individual deltas (0 if none)
 *   neg_ceiling = min of all negative individual deltas (0 if none)
 *   result = clamp(sum, neg_ceiling, pos_ceiling)
 *
 * This prevents stacking multiple emojis to exceed the effect of the
 * strongest single emoji for any given emotion direction.
 */
export function aggregateEmotionEffects(
  effects: EmotionEffect[],
): EmotionEffect[] {
  // Group effects by emotion
  const grouped = new Map<Ekman8Emotion, EmotionEffect[]>();
  for (const effect of effects) {
    const existing = grouped.get(effect.emotion) ?? [];
    existing.push(effect);
    grouped.set(effect.emotion, existing);
  }

  const aggregated: EmotionEffect[] = [];

  for (const [emotion, group] of grouped) {
    const sum = group.reduce((acc, e) => acc + e.delta, 0);

    // Find positive ceiling: highest positive individual delta
    const positiveDeltas = group.map(e => e.delta).filter(d => d > 0);
    const posCeiling = positiveDeltas.length > 0 ? Math.max(...positiveDeltas) : 0;

    // Find negative ceiling: most negative individual delta (smallest value)
    const negativeDeltas = group.map(e => e.delta).filter(d => d < 0);
    const negCeiling = negativeDeltas.length > 0 ? Math.min(...negativeDeltas) : 0;

    // Clamp the sum between the two ceilings
    const clamped = Math.max(negCeiling, Math.min(posCeiling, sum));

    // Only include non-zero results
    if (Math.abs(clamped) > 0.001) {
      aggregated.push({ emotion, delta: Math.round(clamped * 100) / 100 });
    }
  }

  return aggregated;
}

// ============================================================================
// Service Class
// ============================================================================

class EntityEmojiActionService {
  /**
   * In-memory cache: entityId → Map<emojiNative, EmojiAction>
   * Loaded lazily and invalidated on mutations.
   */
  private cache = new Map<string, Map<string, EmojiAction>>();

  /**
   * Get all actions for an entity as a Map (cached).
   * This is the hot path — called on every keystroke in the custom input.
   */
  async getActionsMap(entityId: string): Promise<Map<string, EmojiAction>> {
    let map = this.cache.get(entityId);
    if (map) return map;

    map = await getEmojiActionsForEntity(entityId);
    this.cache.set(entityId, map);
    return map;
  }

  /**
   * Get a single action for an entity+emoji (uses cached map).
   */
  async getAction(entityId: string, emojiNative: string): Promise<EmojiAction | null> {
    const map = await this.getActionsMap(entityId);
    return map.get(emojiNative) ?? null;
  }

  /**
   * Get all actions for an entity as an array.
   * Used by the action editor screen.
   */
  async getAllActions(entityId: string): Promise<EmojiAction[]> {
    return getAllEmojiActionsArray(entityId);
  }

  /**
   * Save (create or update) an emoji action.
   * Invalidates the cache for this entity.
   */
  async saveAction(action: Omit<EmojiAction, 'createdAt' | 'updatedAt'>): Promise<void> {
    await upsertEmojiAction(action);
    this.cache.delete(action.entityId); // Invalidate cache
    log.info(`Saved action for ${action.emojiNative} on entity ${action.entityId}`);
  }

  /**
   * Delete an emoji action by ID.
   * Invalidates the cache for this entity.
   */
  async removeAction(actionId: string, entityId: string): Promise<void> {
    await deleteEmojiAction(actionId);
    this.cache.delete(entityId);
    log.info(`Deleted action ${actionId} from entity ${entityId}`);
  }

  /**
   * Resolve all emoji actions in a message text.
   *
   * 1. Split text into emoji segments using EmojiService.splitTextOnEmojis()
   * 2. For each emoji segment, check if there's an action mapping
   * 3. Build substituted text (replacing emojis with their substitution text)
   * 4. Collect all emotion effects and aggregate them
   *
   * Returns the substituted text, aggregated effects, and whether any actions were applied.
   */
  async resolveMessageActions(
    entityId: string,
    text: string,
  ): Promise<ResolvedMessageActions> {
    const actionsMap = await this.getActionsMap(entityId);
    const segments = EmojiService.splitTextOnEmojis(text);

    const textParts: string[] = [];
    const rawEffects: EmotionEffect[] = [];
    let hasActions = false;

    for (const segment of segments) {
      if (segment.type === 'text') {
        textParts.push(segment.value);
      } else {
        // Emoji segment — check for action
        const action = actionsMap.get(segment.value);
        if (action) {
          hasActions = true;

          // Replace emoji with substitution text
          if (action.substitutionText) {
            textParts.push(action.substitutionText);
          } else {
            // No substitution text — keep the emoji as-is
            textParts.push(segment.value);
          }

          // Collect emotion effect
          if (action.emotionEffect) {
            rawEffects.push(action.emotionEffect);
          }
        } else {
          // No action — keep the emoji as-is
          textParts.push(segment.value);
        }
      }
    }

    // Aggregate emotion effects (clamp per-emotion)
    const aggregatedEffects = aggregateEmotionEffects(rawEffects);

    return {
      substitutedText: textParts.join(''),
      effects: {
        emotionEffects: aggregatedEffects,
      },
      hasActions,
    };
  }

  /**
   * Auto-generate substitution text from the action's effects.
   * Uses simple template-based generation:
   *   - If metabolism vector: "{verb} {item} {emotion_adverb}"
   *   - If emotion only: "feels {emotion_description}"
   */
  generateSubstitutionText(action: {
    emotionEffect: EmotionEffect | null;
    metabolismVector: MetabolismVector | null;
  }): string | null {
    const parts: string[] = [];

    // Metabolism verb + item
    if (action.metabolismVector) {
      const verbMap: Record<string, string> = {
        eat: 'eats',
        drink: 'drinks',
        sleep: 'falls asleep',
      };
      const verb = verbMap[action.metabolismVector.type] ?? action.metabolismVector.type;
      parts.push(`${verb} ${action.metabolismVector.item}`);
    }

    // Emotion adverb
    if (action.emotionEffect) {
      const adverbMap: Record<Ekman8Emotion, string> = {
        joy: 'with enjoyment',
        sadness: 'with a hint of sadness',
        trust: 'trustingly',
        disgust: 'with visible distaste',
        fear: 'fearfully',
        anger: 'angrily',
        surprise: 'in surprise',
        anticipation: 'eagerly',
      };

      const adverb = adverbMap[action.emotionEffect.emotion];

      if (action.metabolismVector) {
        // Combine: "eats a burger with enjoyment"
        if (action.emotionEffect.delta > 0) {
          parts.push(adverb);
        }
      } else {
        // Emotion only: "feels a warm sense of joy"
        if (action.emotionEffect.delta > 0) {
          parts.push(`feels a warm sense of ${action.emotionEffect.emotion}`);
        } else {
          parts.push(`feels a lingering sense of ${action.emotionEffect.emotion} fade`);
        }
      }
    }

    if (parts.length === 0) return null;

    // Wrap in asterisks for RP formatting
    return `*${parts.join(' ')}*`;
  }

  /**
   * Seed default emoji actions for an entity.
   * Called when an entity starts its first session (or on demand via "Reset to Defaults").
   * If actions already exist, only fills in missing defaults (doesn't overwrite user changes).
   *
   * @param force If true, deletes all existing actions before seeding (used for "Reset to Defaults").
   */
  async seedDefaults(entityId: string, force = false): Promise<void> {
    if (force) {
      await deleteAllEmojiActions(entityId);
      this.cache.delete(entityId);
    } else {
      // Only seed if entity has no actions yet
      const count = await countEmojiActions(entityId);
      if (count > 0) {
        log.info(`Entity ${entityId} already has ${count} actions, skipping default seed`);
        return;
      }
    }

    const existingMap = force ? new Map() : await this.getActionsMap(entityId);

    let seeded = 0;
    for (const seed of DEFAULT_ACTION_SEEDS) {
      // Skip if this emoji already has a user-defined action
      if (!force && existingMap.has(seed.emoji)) continue;

      const id = `${entityId}_${seed.emoji}_default`;

      await createEmojiAction({
        id,
        entityId,
        emojiNative: seed.emoji,
        emotionEffect: seed.emotion ?? null,
        metabolismVector: seed.metabolism ?? null,
        substitutionText: seed.substitution,
        autoGenerated: true,
        isDefault: true,
      });

      seeded++;
    }

    // Invalidate cache after seeding
    this.cache.delete(entityId);
    log.info(`Seeded ${seeded} default actions for entity ${entityId}`);
  }

  /**
   * Invalidate the cache for a specific entity.
   * Call after any external mutation (e.g., sync).
   */
  invalidateCache(entityId: string): void {
    this.cache.delete(entityId);
  }

  /**
   * Invalidate all cached data.
   */
  invalidateAllCaches(): void {
    this.cache.clear();
  }
}

// Export singleton instance
export default new EntityEmojiActionService();
